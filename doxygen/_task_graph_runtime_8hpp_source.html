164  std::list<AnyTaskManager *> newVertices;
165  DEBUG_VERBOSE("Launching runtime for " << vertices->size() << " vertices");

168 
169  DEBUG_VERBOSE("Spawning " << numThreads << " threads for task " << task->getName());
170 
171  if (numThreads > 0) {

173  std::shared_ptr<std::atomic_size_t>
174  atomicNumThreads = std::shared_ptr<std::atomic_size_t>(new std::atomic_size_t(numThreads));
175  taskList.push_back(task);
176 
177  for (size_t i = 1; i < numThreads; i++) {
178  AnyTaskManager *taskCopy = task->copy(true);
179 
180 #ifdef WS_PROFILE
181  // Generate . . . and send data . . .
182  std::shared_ptr<ProfileData> producerData(new CreateNodeProfile(taskCopy->getTaskFunction(), graph, taskCopy->getName()));
183  graph->sendProfileData(producerData);
184 
185  std::shared_ptr<ProfileData> connectorConsumerData(new CreateEdgeProfile(taskCopy->getInputConnector().get(), taskCopy->getTaskFunction(), "", nullptr));
186  std::shared_ptr<ProfileData> producerConnectorData(new CreateEdgeProfile(taskCopy->getTaskFunction(), taskCopy->getOutputConnector().get(), "", nullptr));
187 
188  graph->sendProfileData(connectorConsumerData);
189  graph->sendProfileData(producerConnectorData);
190 #endif
191 
192  // Add communicator to task copy to enable communication
193  taskCopy->setTaskGraphCommunicator(graph->getTaskGraphCommunicator());
194  taskList.push_back(taskCopy);
195  newVertices.push_back(taskCopy);
196  }
197  size_t threadId = 0;
198  for (AnyTaskManager *taskItem : taskList) {
199  TaskManagerThread *runtimeThread = new TaskManagerThread(threadId, taskItem, atomicNumThreads);
200  std::thread *thread = new std::thread(&TaskManagerThread::run, runtimeThread);
201  this->threads.push_back(thread);
202  runtimeThreads.push_back(runtimeThread);
203  threadId++;
204  }
205 
206  } else {

208  }
209  }
