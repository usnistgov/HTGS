145  void computeAverages() {
146  std::map<AnyTaskManager *, TaskManagerProfile *> finalProfiles;
147  // Address + name + thread ID = unique
148 
149  // Group using Address + name ... thread ID = 0 is final version
150 
151  std::multimap<std::string, std::pair<AnyTaskManager *, TaskManagerProfile *>> averageMap;
152 
153  std::set<std::string> keys;
154 
155  // Gather multimap
156  for (auto t : *taskManagerProfiles) {
157  auto tMan = t.first;
158  std::string key = tMan->getAddress() + tMan->getName();
159  keys.insert(key);
160  averageMap.insert(std::pair<std::string, std::pair<AnyTaskManager *, TaskManagerProfile *>>(key, t));
161  }
162 
163  // Loop through each key
164  for (auto key : keys) {
165  auto valRange = averageMap.equal_range(key);
166 
167  AnyTaskManager *mainManager = nullptr;
168  TaskManagerProfile *finalProfile = nullptr;
169 
170  int count = 0;
171  for (auto i = valRange.first; i != valRange.second; ++i) {
172  auto profilePair = (*i).second;
173 
174  if (finalProfile == nullptr) {
175  finalProfile = profilePair.second;
176  } else {
177  finalProfile->sum(profilePair.second);
178  }
179 
180  if (profilePair.first->getThreadId() == 0) {
181  mainManager = profilePair.first;
182  } else {
183  delete profilePair.second;
184  profilePair.second = nullptr;
185  }
186  count++;
187  }
188 
189  if (finalProfile != nullptr && mainManager != nullptr) {
190  finalProfile->average(count);
191  finalProfiles.insert(std::pair<AnyTaskManager *, TaskManagerProfile *>(mainManager, finalProfile));
192  } else {
193  std::cout << "Something screwy happened . . ." << std::endl;
194  }
195  }
196 
197  taskManagerProfiles->clear();
198 
199  for (auto t : finalProfiles) {
200  taskManagerProfiles->insert(t);
201  }
202 
203  }

213  std::unordered_map<std::string, std::string> *genColorMap(int colorFlag) {
214  std::unordered_map<std::string, std::string> *colorMap = new std::unordered_map<std::string, std::string>();
215 
216  int rColor[10] = {0, 0, 0, 0, 85, 170, 255, 255, 255, 255};
217  int gColor[10] = {0, 85, 170, 255, 255, 255, 255, 170, 85, 0};
218  int bColor[10] = {255, 255, 255, 255, 170, 85, 0, 0, 0, 0};
219 
220  std::deque<double> vals;
221  double maxTime = 0.0;
222  double totalTime = 0.0;
223  for (auto v : *taskManagerProfiles) {
224  double val = v.second->getValue(colorFlag);
225  if (val > 0) {
226  totalTime += val;
227  }
228 
229  if (maxTime < val)
230  maxTime = val;
231  }
232 
233  for (auto v : *taskManagerProfiles) {
234  if (v.second->getValue(colorFlag) == 0.0) {
235  colorMap->insert(std::pair<std::string, std::string>(v.first->getTaskFunction()->getDotId(), "black"));
236  continue;
237  }
238 
239  int red = 0;
240  int green = 0;
241  int blue = 0;
242 
243  // compute percentage of totalTime
244  int perc = (int) (v.second->getValue(colorFlag) / maxTime * 100.0);
245 
246  if (perc % 10 != 0)
247  perc = perc + 10 - (perc % 10);
248 
249  int index = (perc / 10);
250 
251  if (index < 0)
252  index = 0;
253 
254  if (index >= 10)
255  index = 9;
256

258  green = gColor[index];
259  blue = bColor[index];
260

262

264  std::string color(hexcol);
265  color = "#" + color;
266 
267  colorMap->insert(std::pair<std::string, std::string>(v.first->getTaskFunction()->getDotId(), color));
268  }
269
