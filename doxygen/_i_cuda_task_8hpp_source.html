279  if (requiresCopy(data)) {

283  this->cudaIds[data->getPipelineId()],

296  void initialize() override final {
297  this->cudaId = this->cudaIds[this->getPipelineId()];
298  this->context = this->contexts[this->getPipelineId()];
299 
300  cuCtxSetCurrent(this->context);
301 
302  CUstream stream;
303  cuStreamCreate(&stream, CU_STREAM_DEFAULT);
304 
305  this->stream = stream;
306 
307  CUdevice dev;
308  cuDeviceGet(&dev, this->cudaId);
309 
310  for (int i = 0; i < this->numGpus; i++) {
311  CUcontext ctx = this->contexts[i];
312  if (ctx != this->context) {
313  CUdevice peerDev;
314  cuDeviceGet(&peerDev, this->cudaIds[i]);
315 
316  int canAccessPeer;
317  cuDeviceCanAccessPeer(&canAccessPeer, dev, peerDev);
318 
319  if (canAccessPeer == 0) {
320  this->nonPeerDevIds.push_back(this->cudaIds[i]);
321  this->peerContexts.insert(std::pair<int, CUcontext>(this->cudaIds[i], ctx));
322  } else {
323  cuCtxEnablePeerAccess(ctx, 0);
324  }
325  }
326  }
327 
328  this->initializeCudaGPU();
329  }
