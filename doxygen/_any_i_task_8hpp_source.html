595  template<class V>
596  m_data_t<V> getMemory(std::string name, IMemoryReleaseRule *releaseRule, MMType type, size_t nElem) {
597  assert(("Unable to find memory edge 'name' for task", this->memoryEdges->find(name) != this->memoryEdges->end()));
598 
599  auto conn = memoryEdges->find(name)->second;
600  auto connector = std::dynamic_pointer_cast<Connector<MemoryData<V>>>(conn);
601 
602 #ifdef WS_PROFILE
603  sendWSProfileUpdate(StatusCode::WAITING_FOR_MEM);
604 #endif
605 
606 #ifdef PROFILE
607  auto start = std::chrono::high_resolution_clock::now();
608 #endif
609  m_data_t<V> memory = connector->consumeData();
610 #ifdef PROFILE
611  auto finish = std::chrono::high_resolution_clock::now();
612  this->incMemoryWaitTime(std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count());
613 #endif
614 
615 #ifdef WS_PROFILE
616  sendWSProfileUpdate(StatusCode::EXECUTE);
617 #endif
618 
619 
620 
621  memory->setMemoryReleaseRule(releaseRule);
622 
623  if (memory->getType() != type) {
624  std::cerr
625  << "Error: Incorrect usage of getMemory. Dynamic memory managers use 'getDynamicMemory', Static memory managers use 'getMemory' for task "
626  << this->getName() << " on memory edge " << name << std::endl;
627  exit(-1);
628  }
629 
630  if (type == MMType::Dynamic)
631  memory->memAlloc(nElem);
632 
633  return memory;
634  }
