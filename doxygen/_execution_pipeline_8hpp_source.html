197  for (size_t i = 0; i < numPipelinesExec; i++) {

201  this->getTaskGraphCommunicator());

207  RuleManager<T, T> *ruleManager = new RuleManager<T, T>(rule);
208  ruleManager->setOutputConnector(graphCopy->getInputConnector());
209  ruleManager->initialize(i, this->numPipelinesExec, this->getAddress());
210 
211  this->inputBk->addRuleManager(ruleManager);
212  }
213 
214  graphs->push_back(graphCopy);

216  }
