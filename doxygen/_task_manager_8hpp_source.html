136  void executeTask() override {
137  std::shared_ptr<T> data = nullptr;
138 
139  DEBUG_VERBOSE(prefix() << "Running task: " << this->getName());
140 
141  if (this->isStartTask()) {
142  DEBUG_VERBOSE(prefix() << this->getName() << " is a start task");
143  this->setStartTask(false);
144  auto start = std::chrono::high_resolution_clock::now();
145  this->taskFunction->executeTask(nullptr);
146  auto finish = std::chrono::high_resolution_clock::now();
147 
148  this->incTaskComputeTime(std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count());
149  return;
150  } else if (this->taskFunction->canTerminate(this->inputConnector)) {
151 
152  DEBUG(prefix() << this->getName() << " task function is terminated");
153  this->processTaskFunctionTerminated();
154 
155  return;
156  }
157  auto start = std::chrono::high_resolution_clock::now();
158 
159  if (this->isPoll())
160  data = this->inputConnector->pollConsumeData(this->getTimeout());
161  else
162  data = this->inputConnector->consumeData();
163 
164  auto finish = std::chrono::high_resolution_clock::now();
165 
166  this->incWaitTime(std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count());
167 
168  DEBUG_VERBOSE(prefix() << this->getName() << " received data: " << data << " from " << inputConnector);
169 
170  if (data != nullptr) {
171  start = std::chrono::high_resolution_clock::now();
172  this->taskFunction->executeTask(data);
173  finish = std::chrono::high_resolution_clock::now();
174 
175  this->incTaskComputeTime(std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count());
176  }

232  void processTaskFunctionTerminated() {
233  // Task is now terminated, so it is no longer alive
234  this->setAlive(false);
235 
236  // Wake up the threads for this task
237  this->getInputConnector()->wakeupConsumer();
238 
239  // If there is a runtime thread, then begin termination
240  if (this->runtimeThread != nullptr) {
241  this->runtimeThread->terminate();
242 
243  // If this is the last thread for this task then close the output
244  if (this->runtimeThread->decrementAndCheckNumThreadsRemaining()) {
245  if (this->getOutputConnector() != nullptr) {
246  this->getOutputConnector()->producerFinished();
247 
248  if (this->getOutputConnector()->isInputTerminated())
249  this->getOutputConnector()->wakeupConsumer();
250  }
251 
252  auto memManagerConnectorMap = this->getTaskFunction()->getReleaseMemoryEdges();
253 
254  DEBUG(prefix() << " " << this->getName() << " Shutting down " << memManagerConnectorMap->size()
255  << " memory releasers");
256  for (auto nameManagerPair : *memManagerConnectorMap) {
257  DEBUG(prefix() << " " << this->getName() << " Shutting down memory manager: " << nameManagerPair.first);
258 
259  std::shared_ptr<AnyConnector> connector = nameManagerPair.second;
260  connector->producerFinished();
261 
262  if (connector->isInputTerminated())
263  connector->wakeupConsumer();
264  }
265  }
266  } else {
267  if (this->getOutputConnector() != nullptr) {
268  this->getOutputConnector()->producerFinished();
269 
270  if (this->getOutputConnector()->isInputTerminated()) {
271  this->getOutputConnector()->wakeupConsumer();
272  }
273  }
274  }
