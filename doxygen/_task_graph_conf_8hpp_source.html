220  TaskGraphConf<T, U> *copy(size_t pipelineId,
221  size_t numPipelines,
222  std::shared_ptr<Connector<T>> input,
223  std::shared_ptr<Connector<U>> output,
224  std::string baseAddress,
225  TaskGraphCommunicator *parentCommunicator) {
226  TaskGraphConf<T, U> *graphCopy = new TaskGraphConf<T, U>(pipelineId, numPipelines, baseAddress, parentCommunicator);
227 
228  // Copy the tasks to form lookup between old ITasks and new copies
229  graphCopy->copyTasks(this->getTaskManagers());
230 
231  if (input != nullptr) {
232  graphCopy->setInputConnector(input);
233  }
234 
235  if (output != nullptr) {
236  graphCopy->setOutputConnector(output);
237  }
238 
239  // Copy the graph producer and consumer tasks
240  graphCopy->copyAndUpdateGraphConsumerTask(this->graphConsumerTaskManager);
241  graphCopy->copyAndUpdateGraphProducerTasks(this->graphProducerTaskManagers);
242 
243  for (EdgeDescriptor *edgeDescriptor : *edges) {
244  // Copy the edge, using the graph copy as a reference for where to get task copies
245  EdgeDescriptor *edgeCopy = edgeDescriptor->copy(graphCopy);
246 
247  // Apply the edge on the graph copy
248  edgeCopy->applyEdge(graphCopy);
249 
250  graphCopy->addEdgeDescriptor(edgeCopy);
251  }
252 
253  return graphCopy;
254  }
