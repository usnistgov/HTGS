320  void executeTask() {
321  std::shared_ptr<T> data = nullptr;
322 
323  DEBUG_VERBOSE(prefix() << "Running task: " << this->getName());
324 
325  if (isStartTask) {
326  DEBUG_VERBOSE(prefix() << this->getName() << " is a start task");
327  isStartTask = false;
328  auto start = std::chrono::high_resolution_clock::now();
329  this->taskFunction->executeTask(nullptr);
330  auto finish = std::chrono::high_resolution_clock::now();
331 
332  this->taskComputeTime += std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count();
333  return;
334  } else if (this->taskFunction->isTerminated(this->inputConnector)) {
335 
336 #ifdef PROFILE
337  {
338  std::unique_lock<std::mutex> lock(ioMutex);
339  std::cout << "===================== " << this->getName() << " "<< prefix() << " ===================" << std::endl;
340  std::cout << "COMPUTE TIME: " << taskComputeTime << " us WAIT TIME: " << taskWaitTime << " us" << std::endl;
341 
342  if (this->inputConnector != nullptr) {
343  std::cout << "Input connector: ";
344  this->inputConnector->profileConsume(this->numThreads, true);
345  }
346  if (this->outputConnector != nullptr) {
347  std::cout << "Output connector: ";
348  this->outputConnector->profileProduce(this->numThreads);
349  }
350  this->taskFunction->profileITask();
351  std::cout << "-------------------------- " << this->getName() << " (thread: " << this->threadId << ") -------------------------- " << std::endl << std::endl;
352  }
353 #endif
354 
355 
356  DEBUG(prefix() << this->getName() << " task function is terminated");
357  this->alive = false;
358  this->inputConnector->wakeupConsumer();
359 
360 
361  if (this->runtimeThread != nullptr) {
362  this->runtimeThread->terminate();
363 
364  if (this->runtimeThread->decrementAndCheckNumThreadsRemaining()) {
365  if (this->outputConnector != nullptr) {
366  this->outputConnector->producerFinished();
367 
368  if (this->outputConnector->isInputTerminated())
369  this->outputConnector->wakeupConsumer();
370  }
371 
372  std::shared_ptr<std::unordered_map<std::string, std::shared_ptr<std::vector<std::shared_ptr<BaseConnector>>> >> memReleasers = this->taskFunction->getMemReleasers();
373 
374  DEBUG(prefix() << " " << this->getName() << " Shutting down " << memReleasers->size() <<
375  " memory releasers");
376  for (std::pair<std::string, std::shared_ptr<std::vector<std::shared_ptr<BaseConnector>>> > pair : *memReleasers) {
377 
378 
379  if (this->taskFunction->isMemReleaserOutsideGraph(pair.first))
380  {
381  DEBUG(prefix() << " " << this->getName() << " Shutting down ALL memory releasers : " <<
382  pair.first << " with " << pair.second->size() << " connectors");
383  for (auto connector : *pair.second)
384  {
385  connector->producerFinished();
386 
387 
388  if (connector->isInputTerminated())
389  connector->wakeupConsumer();
390  }
391  }
392  else {
393  DEBUG(prefix() << " " << this->getName() << " Shutting down memory releaser : " <<
394  pair.first << " with " << pair.second->size() << " connectors");
395  std::shared_ptr<BaseConnector> connector = pair.second->at((unsigned long) this->pipelineId);
396  connector->producerFinished();
397 
398 
399  if (connector->isInputTerminated())
400  connector->wakeupConsumer();
401  }
402  }
403 
404  }
405  }
406  else {
407  if (this->outputConnector != nullptr) {
408  this->outputConnector->producerFinished();
409 
410  if (this->outputConnector->isInputTerminated()) {
411  this->outputConnector->wakeupConsumer();
412  }
413  }
414  }
415  return;
416  }
417  auto start = std::chrono::high_resolution_clock::now();
418 
419  if (poll)
420  data = this->inputConnector->pollConsumeData(this->timeout);
421  else
422  data = this->inputConnector->consumeData();
423 
424  auto finish = std::chrono::high_resolution_clock::now();
425 
426  this->taskWaitTime += std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count();
427 
428  DEBUG_VERBOSE(prefix() << this->getName() << " received data: " << data << " from " << inputConnector);
429 
430  if (data != nullptr) {
431  auto start = std::chrono::high_resolution_clock::now();
432  this->taskFunction->executeTask(data);
433  auto finish = std::chrono::high_resolution_clock::now();
434 
435  this->taskComputeTime += std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count();
436  }
